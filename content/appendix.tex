\section{Additional Details}
\label{sec:appendix}

This appendix provides additional details on the algorithms used in \sysname.

\begin{algorithm}[h]
\caption{Example scheduling algorithm}
\label{alg:example}
\begin{algorithmic}[1]
\Procedure{Schedule}{requests $R$, resources $C$}
  \State Sort $R$ by priority
  \For{each request $r \in R$}
    \State Assign $r$ to the least loaded resource in $C$
  \EndFor
  \State \Return assignment
\EndProcedure
\end{algorithmic}
\end{algorithm}
